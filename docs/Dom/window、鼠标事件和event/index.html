window 、鼠标事件和 event

1、window 事件​

window.onload​

dom 和资源加载完毕之后执行的事假​

window.onload 一定会在资源和 dom 都加载完毕之后执行

// 在执行script标签的时候，dom页面还没有加载完毕
window.onload = function () {
  // 等待dom结构和资源加载完毕
  var box = document.querySelector(".box");
  box.onclick = function () {
    console.log(123);
  };
};

window.onresize​

当窗口大小发生改变的时候触发事件​

<style>
      .box {
        width: 5rem;
        height: 100px;
        background: red;
      }
</style>

window.onresize = function () {
  console.log("窗口改变了");
};
//   onresize 响应式布局：
// rem 根据字体的大小，html的字体大小，html的默认字体大小是16px
// rem 布局：1rem 1个根字体大小 默认情况下 1rem=1*16px 是16px

// 动态的改变根字体大小
// 1.获取当前浏览器的宽度 window.innerWidth
setFont();
function setFont() {
  var width = window.innerWidth / 10;
  document.documentElement.style.fontSize = width + "px";
}
window.onresize = function () {
  setFont();
};

window.onscroll​

窗口的滚动事件 在窗口滚动的时候触发​

window.onscroll = function () {
  console.log("滚了");
};

获取滚动时候 卷曲的高度 或者宽度​

  • 卷曲的高度：scrollTop
  • 卷曲的宽度：scrollLeft
window.onscroll = function () {
  var num = document.documentElement.scrollTop;
  console.log(num);
};
/* 
     scrollTop可以获取也可以设置
     设置的时候不需要单位
*/
var btn = document.querySelector(".btn");
btn.onclick = function () {
  document.documentElement.scrollTop = 300;
};
/* 
    scrollTop的设置和文档声明有关
    如果有文档声明：需要通过document.documentElement来设置 或者获取卷曲的高度
    如果没有文档声明 需要通过document.body.scrollTop来获取或者设置
*/

2、鼠标事件​

鼠标左键的单击事件 click​

var btn1 = document.querySelector(".btn1");
/* 
        btn1:事件源
        on：事件的修饰符
        click：事件类型
        fn：事件函数，注意事件函数是一个不执行的函数 点击后才会执行
    */
function fn() {
  console.log("点了");
}
btn1.onclick = fn;

双击事件 dblclick​

var btn2 = document.querySelector(".btn2");
btn2.ondblclick = function () {
  console.log("鼠标双击了");
};

按下事件 mousedown​

var btn3 = document.querySelector(".btn3");
btn3.onmousedown = function () {
  console.log("鼠标按下了");
};

移动事件 mousemove​

var boxEle = document.querySelector(".box1");
boxEle.onmousemove = function () {
  console.log("鼠标移动了");
};

抬起事件 mouseup​

var btn4 = document.querySelector(".btn4");
btn4.onmouseup = function () {
  console.log("鼠标抬起了");
};

移入事件 mouseover 和 mouseenter​

// mouseover
var box1Ele = document.querySelector(".box1");
box1Ele.onmouseover = function () {
  console.log("鼠标覆盖了");
};

// mouseenter
var box2Ele = document.querySelector(".box2");
box2Ele.onmouseenter = function () {
  console.log("鼠标进入的时候触发");
};

移出事件 mouseout he mouseleave​

// mouseout
box1Ele.onmouseout = function () {
  console.log("鼠标移出了");
};

// mouseleave
box2Ele.onmouseleave = function () {
  console.log("鼠标离开的时候触发");
};
mouseover/mouseout 和 mouseenter/mouseleave 的区别
mouseover/mouseout 事件不仅会在事件源上触发，在其子元素上也会触发
mouseenter/mouseleave 事件仅会在事件源上触发

滚轮事件 document.onwheel​

document.onwheel = function () {
  console.log("滚了");
};

右键点击 contextmenu​

document.oncontextmenu = function () {
  console.log("右键了");
};

3、事件对象​

事件对象​

事件触发的时候 js 会提供一个对象，这个对象里描述了和事件相关的一些信息

var boxEle = document.querySelector(".box");
boxEle.onclick = function (e) {
  // 通过事件函数的形参接收事件对象
  console.log("clientX", e.clientX);
  console.log("clientY", e.clientY);
  console.log("offsetX", e.offsetX);
  console.log("offsetY", e.offsetY);
};

事件对象的参数​

  • altKey 这个时候是否按了 alt 键
  • button：0 左键 1 滚轮 2 右键
  • clientX: 鼠标触发事件地方到浏览器左边的距离
  • clientY：鼠标触发事件地方到浏览器上边的距离
  • offsetX:鼠标点击事件的时候距离事件源的左边的位置
  • offsetY:鼠标点击事件的时候距离事件源的上边的位置
  • pageX：鼠标触发事件相对于页面的左边的距离
  • pageY：鼠标触发事件相对于页面的上边的距离

事件对象兼容问题​

document.querySelector(".box").onclick = function (e) {
  // 主流浏览器 ：ie8以上浏览器 e
  //     // 主流浏览器 ： IE8 以上浏览器  e
  //     // IE8 及一下浏览器  事件对象的获取  window.event
  var e = e || window.event; // 兼容 IE8及一下浏览器的写法
  //     console.log(e);
};
