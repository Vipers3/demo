元素的尺寸和浏览器信息

1、元素的尺寸和偏移量​

获取元素的尺寸​

offsetWidth 和 offsetHeight​

  • 自身的宽高+内边距+边框的宽或高
var box = document.querySelector(".box");
console.log(box.offsetWidth);
console.log(box.offsetHeight);
  • clientWidth 和 clientHeight
  • 自身宽高+内边距
console.log(box.clientHeight);
console.log(box.clientWidth);

元素的偏移量​

offsetLeft 和 offsetTop​

  • 第一种情况： 没有定位的情况 相对于浏览器的左边和上边的偏移量
  • 第二种情况： 有定位的情况 元素的偏移量会相对于父级的定位元素 左边和上边的偏移量
console.log(box.offsetLeft);

clientLeft 和 clientTop​

  • 获取元素的左边和上边边框的大小：border 的大小
console.log(box.clientTop);
console.log(box.clientLeft);

2、 getBoundingClientRect()获取元素尺寸​

<div class="container">
  <div class="box"></div>
</div>
var box = document.querySelector(".box");
var posObj = box.getBoundingClientRect();

width 和 height​

获取元素的宽和高 自身+padding+border 的宽或者高的大小

console.log(posObj.width, posObj.height);

x/left 和 y/top​

获取元素相对视口的左边和上边的偏移量

console.log(posObj.left, posObj.top);

right 和 bottom​

相对视口+自身+padding+border 的右边和下边的偏移量

console.log(posObj.right, posObj.bottom);

3、获取浏览器尺寸​

DOM:和 document 有关的内容，和文档有关的内容 顶级对象 document

BOM：browser object model 浏览器对象模型，和浏览器操作有关 顶级对象 window

获取浏览器视口的宽和高​

<style>
  .box {
    height: 1000px;
  }
</style>

<div class="box"></div>
// 1.window.innerWidth:获取浏览器视口的宽度
//   innerWidth是包含滚动条的
console.log(window.innerWidth);
//   获取不包含滚动条的宽度
console.log(document.documentElement.clientWidth);

//2.window.innerHeight:获取浏览器视口的高度
console.log(window.innerHeight);

4、浏览器的相关信息 navigator​

window.navigator​

  • appName 浏览器的公司 Netscape
  • appVersion 浏览器的版本
  • language "zh-Ch"
  • platform 浏览器运行的系统
  • userAgent 浏览器型号的一些信息

5、和浏览器地址栏关的信息​

打开一个页面​

window.open("打开页面的地址"，"设置打开方式：\_self 在当前窗口打开页面 \_blank 新打开一个窗口","设置打开窗口的大小和位置")​

<button class="btn1">点我打开一个页面</button>;

var btn1 = document.querySelector(".btn1");
btn1.onclick = function () {
  // window.open("https://www.baidu.com/");
  window.open("./a.html", "_black", "width=200,height=200,left=300,top=200");
  //  可以设置打开页面的大小
};

关闭页面​

window.close( )​

注意：window.close()关闭窗口 只能关闭直接打开的或者是通过 window.open( )打开的

获取浏览器的地址​

window.location.href​

console.log(window.location.href);

设置地址栏信息​

window.location.href = "地址"​

var btn = document.querySelector(".btn");
btn.onclick = function () {
  window.location.href = "https://www.baidu.com/";
};

刷新页面​

window.location.reload( )​

var btn2 = document.querySelector(".btn2");
btn2.onclick = function () {
  window.location.reload(); //刷新页面
  // 不要把刷新页面直接写在js里
  //   会一直刷新死循环
};

获取查询字符串​

window.location.search​

  • 获取地址栏的？号后面的参数 参数是 querystring 的结构
  • querystring 的结构 ：键名 1=值 1&键名 2=值 2&键名 3=值 3
console.log(window.location.search);

哈希参数​

  • 哈希参数：地址栏#后面的内容
  • 无刷路由 不需要刷新页面切换地址
// 利用哈希参数实现无刷页面
var btns = document.querySelectorAll("button");
btns.forEach((item, key) => {
  item.onclick = function () {
    window.location.hash = "page" + key;
    renderDom();
  };
});
function renderDom() {
  var hashStr = window.location.hash;
  var showEle = document.querySelector(".show");
  switch (hashStr) {
    case "#page0":
      showEle.innerHTML = "页面第一次的内容";
      break;
    case "#page1":
      showEle.innerHTML = "页面第二次的内容";
      break;
    case "#page2":
      showEle.innerHTML = "页面第三次的内容";
      break;
  }
}

6、和浏览器历史记录有关的的信息​

前进一个历史记录​

window.history.forward();​

var btn1 = document.querySelector(".btn1");
btn1.onclick = function () {
  window.history.forward();
};

回退一个历史记录​

window.history.back()​

// 和浏览器历史记录有关的内容
// 注意：历史记录的操作的前提是必须有历史记录
var btn2 = document.querySelector(".btn2");
btn2.onclick = function () {
  window.history.back();
};

前进或者回退到指定历史记录：​

window.history.go()​

go 可以接收参数：1.0 是当前页面 2.正数就是前进历史记录 3.负数就是后退几个历史记录

var btn2 = document.querySelector(".btn2");
btn2.onclick = function () {
  window.history.go(2);
};

浏览器当前历史记录的数量​

window.history.length

console.log(window.history.length);

7、定时器​

倒计时定时器​

setTimeout​

只会执行一次

// setTimeout(需要倒计时执行的回调函数，倒计时的时间单位是毫秒)
setTimeout(() => {
  console.log("回调函数里的代码执行了");
}, 2000);

间隔定时器​

setInterval​

间隔一段时间重复执行回调函数里的代码

setInterval(() => {
  console.log("回调函数执行了");
}, 2000);

定时器的注意事项​

1.定时器的执行顺序问题​

定时器里的任务都是异步任务 ，无论时间长短，一定会在同步任务之后执行异步任务

2.定时器里的 this 问题​

var btn = document.querySelector("button");
btn.onclick = function () {
  var that = this; //通过变量把外部的thi保存起来
  setTimeout(function () {
    // 利用一个变量把外部的this保存起来 使用外部的this
    console.log(that);
  }, 2000);
};

3.setTimeout 可以模拟 setInterval​

//   可以通过倒计时定时器模拟间隔定时器
var btn = document.querySelector("button");
btn.onclick = function () {
  // 递归函数
  function fn() {
    setTimeout(() => {
      console.log("执行了");
      fn();
    }, 1000);
  }
  fn();
};

8、获取非行间样式​

获取非行间样式：getComputedStyle(获取样式的元素,"伪类").样式名​

<style>
  div {
    width: 100px;
    height: 100px;
    background: red;
  }
  div::after {
    content: "";
    width: 50px;
    height: 20px;
  }
</style>

<div class="box">我是div</div>
var box = document.querySelector(".box");
var style = getComputedStyle(box, "after");
console.log(style);
