节点的关系和操作

1、子关系和父关系​

<div class="container">
  <span>我是元素</span>
  <p>w我是p标签</p>
  <!-- 我是注释 -->
</div>
<button class="btn1">点击获取子节点</button>
<button class="btn2">点击获取子元素</button>
<button class="btn3">点击获取父节点</button>
<button class="btn4">点击获取父元素</button>

子关系​

获取子节点​

元素.childNodes

var div = document.querySelector(".container");
var btn1 = document.querySelector(".btn1");
btn1.onclick = function () {
  var nodes = div.childNodes;
  console.log(nodes); //获取到的是一个伪数组
  // 包含文本注释
};

获取子元素​

元素.children 伪数组

var btn2 = document.querySelector(".btn2");
btn2.onclick = function () {
  var nodes = div.children;
  console.log(nodes); //伪数组
};

父关系​

获取父节点​

元素.parentNode

var btn3 = document.querySelector(".btn3");
btn3.onclick = function () {
  var nodes = div.parentNode;
  console.log(nodes); //body
};

获取父元素​

元素.parentElement

var btn4 = document.querySelector(".btn4");
btn4.onclick = function () {
  var nodes = div.parentElement;
  console.log(nodes); //body
};
注意：父节点和父元素是一样的

元素.firstElementChild

var btn6 = document.querySelector(".btn6");

};

获取最后一个子节点​

元素.lastChild

<ul>

下一个兄弟节点​

元素.nextSibling

var twoLi = document.querySelector(".active");
var btn1 = document.querySelector(".btn1");
btn1.onclick = function () {
  console.log(twoLi.nextSibling); //#text
  //获取下一个兄弟节点
  // 换行也是一个文本节点
};

下一个兄弟元素​

元素.nextElementSibling

var btn2 = document.querySelector(".btn2");
btn2.onclick = function () {
  console.log(twoLi.nextElementSibling); // li
};

元素.previousSibling

var btn3 = document.querySelector(".btn3");
btn3.onclick = function () {
  console.log(twoLi.previousSibling); //获取上一个兄弟节点
};

上一个兄弟元素​

元素.previousElementSibling

3.元素的操作(增删改查)​

<div class="container">
  <div class="box">我是div1</div>
  <span>hahah</span>
</div>
<button class="btn1">增加</button>
<button class="btn2">删除</button>
<button class="btn3">删除</button>

新增一个元素​

/* 
    添加元素对象的方法 父元素.appendChild(元素对象)
    会在父元素最后 追加元素对象 不会和innerHTML一样 覆盖内容
    */
//    创建一个元素对象
var div2 = document.createElement("div");
div2.classList.add("box2");
div2.innerText = "我是div2";
//   把创建的元素对象放在指定位置
var div = document.querySelector(".container");
var btn1 = document.querySelector(".btn1");
btn1.onclick = function () {
  div.appendChild(div2);
};

删除一个元素​

删除子元素​

父元素.removeChild(子元素对象)

var span = document.querySelector("span");
document.querySelector(".btn2").onclick = function () {
  div.removeChild(span);
};

删除自身​

元素.remove()

var btn3 = document.querySelector(".btn3");
btn3.onclick = function () {
  div.remove();
};

替换元素​

<ul>
  <li>111</li>
  <li class="active">222</li>
  <li>333</li>
  <button class="btn1">替换</button>
  <button class="btn2">添加</button>
  <button class="btn3">添加</button>
</ul>
;

replaceChild("新元素对象", "")​

var btn1 = document.querySelector(".btn1");
var ul = document.querySelector("ul");
var activeLi = document.querySelector(".active");
btn1.onclick = function () {
  // 创建一个新元素对象 替换之前的元素
  var newLi = document.createElement("li");
  newLi.innerText = "替换的li";
  ul.replaceChild(newLi, activeLi);
};

添加元素对象​

  • 父元素.appendChild(元素对象) 在父元素的最后追加一个元素对象
  • 父元素.insertBefore("新元素对象"，"")
var btn2 = document.querySelector(".btn2");
btn2.onclick = function () {
  ul.insertBefore("newLi", "activeLi");
};

克隆一个元素​

  • var 复制的元素=元素.cloneNode()
  • 注意：cloneNode()里可以给布尔值参数，默认是 false，不会复制子节点
var btn3 = document.querySelector(".btn3");
var clone = activeLi.cloneNode(true);
btn2.onclick = function () {
  ul.appendChild(clone);
};
