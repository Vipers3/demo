token 和本地存储

存储 session 的方式​

​ $ npm i express-session

// 1.1 引入处理session的模块
const session = require("express-session");
// 1.2 引入存储到数据库中的中间件
const mongo = require("connect-mongo");

// 1.2 注册处理session的中间件
app.use(
  session({
    // 保存到客户端的键
    name: "sid",
    // 用于给id加密时混淆的一些字段，随便写
    secret: "atguigu",
    // 是否为每次请求设置cookie
    saveUninitialized: false,
    // 是否为每次请重新保存cookie
    resave: true,
    // cookie 的有效期设置
    cookie: {
      // 只能在http请求中使用
      httpOnly: true,
      // 最大的有效时长
      maxAge: 15 * 1000,
    },
    // 一般在项目中，还可以往数据库中存储
    store: mongo.create({
      mongoUrl: "mongodb://127.0.0.1:27017/guigu",
    }),
  })
);

加密技术​

​ 客户端请求数据存储到数据库，防止企业后台员工恶意窃取用户信息，需要加密数据

​ 防止在传输过程中验证或者非法闯入，以特殊形式获取明文传输的敏感数据

密码加密​

​ 下载密码加密的模块 $ npm i bcryptjs -S

// 1.2 引入加密的模块
const bcrypt = require("bcryptjs");

// 2.1.1 创建加密的数位和盐
const salt = bcrypt.genSaltSync(10);
// 2.1.2 获取密码字段，并传入盐得到加密之后的字段
const pwd = bcrypt.hashSync(req.body.pwd, salt);

// 验证：返回用户输入的密码和数据库请求的加密密码
const bool = bcrypt.compareSync(pwd, data.pwd);

token 令牌加密​

​ 注册中间件 $ npm i express-jwt -S

​ token 是服务器端生成，客户端需要手动存储，下次请求手动携带

// 1.3 引入token的中间件
const exJwt = require("express-jwt");

// 2.5 注册token的中间件 - 在跨域之后注册
express.secretKey = "token_guigu no.1017";
app.use(
  exJwt
    .expressjwt({
      // 指定混淆的字段
      secret: express.secretKey,
      // 指定加密的方式
      algorithms: ["HS256"],
    })
    // 指定排除的接口
    .unless({
      path: [/^\/api\//],
    })
);

// 2.5 错误处理中间件
app.use((err, req, res, next) => {
  // 2.5.1 未捕获到token的字段属性
  if (err.name === "UnauthorizedError") {
    return res.send({
      staus: 401,
      message: "无效的token",
    });
  }
  res.send({
    status: 500,
    message: "未知的错误",
  });
});

​ 客户端登录成功之后，产生加密的 token $ npm i jsonwebtoken -S

// 1.3 引入生成token
const jwt = require("jsonwebtoken");

// 生成一个token给客户端携带回去
const token = jwt.sign(
  // 存储的信息
  { users: data },
  // 秘钥
  express.secretKey,
  // 设置过期时间
  { expiresIn: "30s" }
);

res.send({
  code: 200,
  msg: "登录成功!",
  token: "Bearer " + token,
});

​ 客户端手动携带

$.ajax({
  method: "get",
  url: "http://127.0.0.1:3000/users",
  headers: {
    Authorization: localStorage.getItem("token"),
  },
  success(res) {
    console.log(res);
  },
});

跨域和 cookie​

​ 跨域之后 cookie 访问受限 cors 都解决不了, 需要另外设置

// 服务器端设置
app.all("*", (req, res, next) => {
    // 划重点
    res.header("Access-Control-Allow-Credentials", true);
    res.header("Access-Control-Allow-Origin", req.headers.origin);
    res.header("Access-Control-Allow-Headers", "Content-Type,Content-Length,
    res.header("Access-Control-Allow-Methods", "PUT,POST,GET,DELETE,OPTIONS")
    res.header("X-Powered-By", ' 3.2.1')
    next();
})
// 客户端设置
$.ajax({
  method: "post",
  url: "http://127.0.0.1:3000/api/login",
  data: JSON.stringify(data),
  contentType: "application/json",
  // // 添加跨域的识别信息
  xhrFields: {
    withCredentials: true,
  },
  // 影响访问的域
  crossDomain: true,
  success(res) {
    console.log(res);
    // 存储到前端本地
    localStorage.setItem("token", res.token);
  },
});

ajax 的其他事件​

​ 在项目中，经常会出现超时等待，或者手动结束，或者其他辅助工具使用

超时属性​

​ 超过规定的时长，就放弃请求

$.ajax({
  method: "get",
  url: "http://127.0.0.1:3000/users",
  // 超时属性，如果超过时长，就自动取消
  timeout: 3000,
  success(res) {
    console.log(res);
  },
});

手动取消​

​ 当任务处理时间较长未反应，可以手动结束

const xhr = $.ajax({});

$("#cancel").click(() => {
  // 取消请求
  xhr.abort();
});

进度条展示​

​ 一般项目中，都会有一个进度条，用于展示当前请求的进度

​ 以后的工程化项目都需要下载 $ npm i nprogress -S， 目前找 cdn 资源就可以了

<!-- CSS 资源展示进度条 -->
<link
  href="https://cdn.bootcdn.net/ajax/libs/nprogress/0.2.0/nprogress.min.css"
  rel="stylesheet"
/>
<!-- js 的功能 -->
<script src="https://cdn.bootcdn.net/ajax/libs/nprogress/0.2.0/nprogress.min.js"></script>
$.ajax({
  beforeSend() {
    // 请求发出之前
    console.log("请求开始发送");
    // 展示进度条
    NProgress.start();
  },
  complete() {
    // 不管是成功还是失败，最终都会完成这个
    console.log("请求完成啦");
    // 终止展示
    NProgress.done();
  },
});

http 和 https​

​ 都是 超文本传输协议，都是作用在浏览器上的一个应用层协议

​ https 是基于 http 的再次加密，会更加安全，本质上还是一个东西

1. 基于HTTP协议，通过SSL或TLS提供加密处理数据、验证对方身份以及数据完整性保护
    收方能够证实发送方的真实身份；
    发送方事后不能否认所发送过的报文；
    收方或非法者不能伪造、篡改报文。
2. SSL证书需要购买申请，功能越强大的证书费用越高
3. 使用HTTPS协议会使页面的加载时间延长近50%，增加10%到20%的耗电
4. HTTPS连接服务器端资源占用高很多，支持访客多的网站需要投入更大的成本
5. HTTPS协议握手阶段比较费时，对网站的响应速度有影响，影响用户体验
