mongoose

连接数据库​

创建连接的方式​

​ $ npm i mongoose -S

// 1. 引入数据库模块
const mongoose = require("mongoose");

// 2. 创建连接
/*
  mongodb://   数据库连接协议
  :27017       数据库的端口
  /web         数据库的名称
*/
mongoose.set("strictQuery", true);
const uri = "mongodb://127.0.0.1:27017/web";
mongoose.connect(uri, { useNewUrlParser: true });

// 3. 监听连接成功的回调
mongoose.connection.on("open", () => {
  console.log("连接成功");
});

操作集合和文档​

​ 在 mongoose 中，集合和文档可以理解为数组和对象的关系。

​ 但是 mongoose 要求，要先声明集合的类型和注册模型之后，创建的实例才可以操作文档；

// 3. 监听连接成功的回调
mongoose.connection.on("open", () => {
  console.log("连接成功");

  // 3.1 创建集合的规范
  const BookSchma = mongoose.Schema({
    // 配置的是将来这个集合中的数据，长什么样子
    // 书籍名称： 字符类型
    name: String,
    // 书籍价格： 数值类型
    price: Number,
    // 是否上架： 布尔类型
    isSell: Boolean,
    // 所属分类： 数组类型
    type: Array,
    // 发布日期： 日期类型
    publish: Date,
  });

  // 3.2 注册规范的模型 - 返回一个实例, 有s就不再加s，没有s会自动带一个s
  const books = mongoose.model("gaohu", BookSchma);

  // 3.3 添加内容 - 创建文档
  // 添加多条数据 - insertMany   添加单条数据  create
  // 参数1：数组形式符合规范的数据，参数2：回调函数
  books.insertMany(
    [
      {
        name: "《三国演义》",
        price: 29.1,
        isSell: true,
        type: ["经典", "名著"],
        publish: new Date("2021-10-01"),
      },
      {
        name: "《西游记》",
        price: 19.8,
        isSell: true,
        type: ["经典", "名著"],
        publish: new Date("2001-09-01"),
      },
      {
        name: "《JavaScript指南》",
        price: 108,
        isSell: true,
        type: ["计算机", "前端开发"],
        publish: new Date("2013-02-06"),
      },
      {
        name: "《哈利波特1》",
        price: 9.9,
        isSell: false,
        type: ["奇幻", "魔法"],
        publish: new Date("2009-08-08"),
      },
      {
        name: "《活着》",
        price: 50,
        isSell: false,
        type: ["文学", "小说"],
        publish: new Date("2002-08-16"),
      },
    ],
    (err, data) => {
      if (err) throw err;

      console.log("数据添加成功:", data);

      // 可选：
      mongoose.connection.close();
    }
  );
});

封装连接的操作​

​ 之前的操作都是嵌套在函数中，而且固定在一个页面里面，并且不能让接口内部进行操作

​ 封装之后，会抽离很多的逻辑，这样复用性会更高，代码能够更加清晰

基本封装​

​ 先把代码抽离出来让功能一样的实现

​ 主要实现 3 个大逻辑，

​ 1 是连接数据库的封装

​ 2 规范声明和模型注册

​ 3 调用连接和增删改查的操作

// 创建 db 文件夹，创建 index.jsx  内部完成数据库的连接和回调操作

// 1. 引入数据库模块
const mongoose = require("mongoose");

// 2.1 创建连接信息
const URI = `mongodb://127.0.0.1:27017/guigu`;

// 2.2 连接数据库的操作
mongoose.set("strictQuery", true);
mongoose.connect(URI, { useNewUrlParser: true });

// 3. 创建监听事件
module.exports = function (success, error) {
  mongoose.connection.on("open", () => {
    // 当成功之后，执行的是回调函数
    success && success();
  });

  mongoose.connection.on("error", () => {
    error && error();
  });
};
// 创建 model 文件夹， 创建 users.js 文件用于管理规范和集合

// 1. 引入核心模块
const mongoose = require("mongoose");

// 2. 声明规范 Schema
const userSchema = mongoose.Schema({
  // 名称
  name: String,
  // 年龄
  age: Number,
  // 地址
  addr: String,
  // 性别
  gender: Boolean,
  // 爱好
  hobby: Array,
  // 生日
  birth: Date,
});

// 3. 注册规范，产生模型实例
module.exports = mongoose.model("users", userSchema);
// 直接在路由文件中，调用连接方法，并且操作模型的增删查改

// 测试连接的代码
const connect = require("../db/index.jsx");
const users = require("../model/users.js");
connect(
  // 成功的回调
  () => {
    console.log("success, 连接ok~");
    // 使用模型，操作数据库的增删查改
    users.create(
      {
        name: "迪丽热巴",
        age: 22,
        addr: "新疆·乌鲁木齐",
        gender: false,
        hobby: ["唱", "跳", "rap", "篮球"],
        birth: new Date("2000-10-02"),
      },
      (err, data) => {
        if (err) throw err;
        console.log(data);
      }
    );

  }

// 3. 创建监听事件
module.exports = function (success, error) {
  // 每次请求的时候，就需要重新连接
  // mongoose.connect(URI, { useNewUrlParser: true });
  // mongoose.connection.on('open', () => {
  //     // 当成功之后，执行的是回调函数
  //     success && success();
  // })
  // mongoose.connection.on('error', () => {
  //     error && error();
  // })

  // 参数1: 连接信息  参数2：成功的回调  参数3 失败的回调
  // mongoose.connect(URI, () => {
  //     success && success();
  // }, () => {
  //     error && error();
  // })

  // 更高级的写法
  mongoose
    .connect(URI, { useNewUrlParser: true })
    .then(() => {
      // ES2020 提出了一个 链式判断调用的运算符
      success?.();
    })
    .catch(() => {
      error?.();
    });
};

​ 后端接口

// 3. 添加成员的接口
router.post("/adduser", (req, res) => {
  // 3.1 很友好的得到了一个处理之后的对象，但是年龄，性别，生日数据类型不一致
  const { gender, age, birth } = req.body;
  req.body.gender = gender === "男" ? true : false;
  req.body.age = age - 0;
  req.body.birth = new Date(birth);

  // 3.2 重新连接数据库
  connect(() => {
    // 3.3 使用集合的模型实例操作增加功能
    users.create(req.body, (err) => {
      if (err) throw err;

      res.send({
        code: 200,
        msg: "成员添加成功！",
      });
    });
  });
});

编辑用户​

// 根据地址栏的id发起请求
getData();
function getData() {
  const id = location.search.split("=")[1];
  if (!id) return;
  $.ajax({
    method: "get",
    url: "/getUser?id=" + id,
    success(res) {
      // console.log(res);
      // 数据的回写
      if (res.code === 200) {
        // 从res从解构出来data，再从data中解构出来其他字段
        const {
          data: { name, age, addr, gender, hobby, birth },
        } = res;
        $('input[name="name"]').val(name);
        $('input[name="age"]').val(age);
        $('input[name="birth"]').val(moment(birth).format("YYYY-MM-DD"));
        $('[name="addr"]').val(addr);
        // 性别
        const n = gender ? 0 : 1;
        $(".gender input").eq(n).prop("checked", true);
        // 爱好
        $(".hobby input").each((i, item) => {
          // 从已知的结果中进行判断，是否含有此表单的val值
          if (hobby.includes($(item).val())) {
            $(item).prop("checked", true);
          }
        });
      }
    },
  });
}
// 5. 根据id获取指定的成员

      res.send({
        code: 200,
        msg: "查询成功！",
        data,
      });
    });
  });
});
