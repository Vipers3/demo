promise的实现

Promise 的封装​

1. 基本功能​

// 1. 声明构造函数
function Promise(executor) {
  // 3. 定义默认状态和回调的两个处理状态的函数
  this.PromiseState = "pendding";
  this.PromiseResult = undefined;

  const success = (val) => {
    this.PromiseState = "fulfilled";
    this.PromiseResult = val;
  };
  const error = (err) => {
    this.PromiseState = "rejected";
    this.PromiseResult = err;
  };

  // 2. 同步执行Promise的回调
  try {
    executor?.(success, error);
  } catch (err) {
    error(err.message || err);
  }
}
new Promise(function (resolve, reject) {
  resolve("hello");
  // 改变状态有3种形式
});

2. 实例方法​

// 1. 给原型挂载方法
Promise.prototype.then = function (onFulfilled, onRejected) {
  if (this.PromiseState === "fulfilled") {
    onFulfilled(this.PromiseResult);
  }

  if (this.PromiseState === "rejected") {
    onRejected(this.PromiseResult);
  }
};
p1.then(
  (data) => {},
  (err) => {}

3. 异步处理​

function Promise() {
  // 定义回调函数的队列
  this.callbacks = [];
}
Promise.prototype.then = function (onFulfilled, onRejected) {
  // 实例调用then方法的时候，状态还没有改变，因为是异步操作
  // 因为状态还没有改变，所以状态默认pendding
  // 也可以说只要then方法调用的时候是pendding状态，则是异步操作

  // 因为不确定状态，所以这里的参数回调一个也不能调用
  // 不管是成功还是失败，最终还是会确定的，也就是构造函数中的状态改变会指定
  if (this.PromiseState === "pendding") {
    this.callbacks.push({
      ok() {
        onFulfilled();
      },
      ng() {
        onRejected();
      },
    });
  }
};
function Promise() {
  const success = (val) => {
    // .. 改变状态

    if (this.callbacks.length > 0) {
      this.callbacks.forEach((item) => {
        item.ok();
      });
    }
  };

  const error = (err) => {
    // .. 改变状态

    if (this.callbacks.length > 0) {
      this.callbacks.forEach((item) => {
        item.ng();
      });
    }
  };
}

4. 代码优化​

// 1. 状态是不可逆的
function Promise(executor) {
  const success = (val) => {
    if (this.PromiseState !== "pendding") return;

    this.PromiseState = "fulfilled";
    this.PromiseResult = val;
  };
  const error = (err) => {
    if (this.PromiseState !== "pendding") return;

    this.PromiseState = "rejected";
    this.PromiseResult = err;
  };
}
// 2. then 方法中的 onFulfilled 和 onRejected 是异步的
Promise.prototype.then = function (onFulfilled, onRejected) {
  if (this.PromiseState === "fulfilled") {
    // 定时器一定要写箭头函数，不然this会出现问题
    setTimeout(() => {
      onFulfilled(this.PromiseResult);
    });
  }

  if (this.PromiseState === "rejected") {
    setTimeout(() => {
      onRejected(this.PromiseResult);
    });
  }
};
function Promise() {
  const success = (val) => {
    if (this.callbacks.length > 0) {
      this.callbacks.forEach((item) => {
        setTimeout(() => item.ok());
      });
    }
  };

  const error = (err) => {
    if (this.callbacks.length > 0) {
      this.callbacks.forEach((item) => {
        setTimeout(() => item.ng());
      });

  };
}

5. then 返回值​

/*
    a. then 方法会返回一个 Promise的实例
    b. 如果 then 方法中的回调有返回值
        回调的返回值非 Promise 实例，会给 then 返回值 promise 的结果
        回调的返回值是 Promise 实例，会给 then 返回值状态和对应结果
        回调内报错了，会捕获错误并且标记状态
*/
// 核心，要关注的是谁在操作 then 方法的回调
Promise.prototype.then = function (onFulfilled, onRejected) {
  return new Promise((resolve, reject) => {
    // 封装的返回值的处理
    const handler = (type) => {
      try {
        const returns = type(this.PromiseResult);
        if (returns instanceof Promise) {
          // 如果是回调的返回值是Promise实例
          // 要先确定这个实例的状态是成功还是失败
          // 获取到成功或者失败的值，再给then返回值加上去
          returns.then(
            (data) => resolve(data),
            (err) => reject(err)
          );
        } else {
          resolve(returns);
        }
      } catch (err) {
        reject(err);
      }
    };

    if (this.PromiseState === "pendding") {
      this.callbacks.push({
        ok() {
          handler(onFulfilled);
        },
        ng() {
          handler(onRejected);
        },
      });
    }

    if (this.PromiseState === "fulfilled") {
      handler(onFulfilled);
      /*
            try {
                const returns = onFulfilled()
                if(returns instanceof Promise) {
                    // 如果是回调的返回值是Promise实例
                    // 要先确定这个实例的状态是成功还是失败
                    // 获取到成功或者失败的值，再给then返回值加上去
                    returns.then(data => resolve(data), err => reject(err))
                } else {
                    resolve(returns)
                }
            } catch(err) {
                reject(err)
            }
            */
    }

    if (this.PromiseState === "rejected") {
      handler(onRejected);
    }
  });
};

6. catch 方法​

/*
    a. 会返回promise的实例
    b. 实例的状态和结果由形参调用的返回值决定
    c. 异步的处理和then方法一致
*/
Promise.prototype.catch = function (onRejected) {
  return this.then(undefined, onRejected);
};
// 情景一： then 提前处理错误了，导致到catch的时候，是成功状态的 promise 实例
const res = new Promise((resolve, reject) => {
  reject("error 错误信息111");
})
  .then(
    (data) => {},
    (err) => {
      console.log(err, "提前处理失败的");
    }
  )
  .catch((err) => {
    console.log(err, "成功处理");
  });
// 弥补在先处理错误，导致catch没有带成功参数的情况
if (typeof onFulfilled !== "function") {
  // 补一个成功的回调
  onFulfilled = (v) => v;
}
// 情景二: then 方法在设计的时候要求传入两个回调，但是只传入了一个
const res = new Promise((resolve, reject) => {
  reject("error 错误信息111");
})
  // 因为上面的状态是失败，在then方法内部缺省回调，导致报错，得到的信息不一致
  .then((data) => {})
  .catch((err) => {
    console.log(err, "成功处理");
  });
if (typeof onRejected !== "function") {
  // 补一个失败的回调, 上述的状态修改什么值，则接收并继续改变错误的状态带什么值
  onRejected = (err) => {
    throw err;
  };
}

7. 静态方法​

// 3. 静态方法
Promise.resolve = function (data) {
  return new Promise((resolve, reject) => {
    // 只需要判断data是否为Promise实例
    if (data instanceof Promise) {
      // data.then(resolve, reject)
      data.then(
        (val) => resolve(val),
        (error) => reject(error)
      );
    } else {
      resolve(data);
    }
  });
};

Promise.reject = function (val) {
  return new Promise((resolve, reject) => {
    reject(val);
  });
};
// race 方法
Promise.race = function (taskArr) {
  return new Promise((resolve, reject) => {
    // 遍历任务
    taskArr.forEach((item) => {
      item.then(
        (data) => resolve(data),
        (err) => reject(err)
      );
    });
  });
};
// 定义多个任务
const task1 = new Promise((resolve, reject) => {
  setTimeout(() => {
    resolve("成功了1");
  }, 2000);
});
const task2 = Promise.reject("数据回来了2");
const task3 = new Promise((resolve, reject) => {
  resolve("非常OK3");
});
// 调用race 方法
const res = Promise.race([task1, task2, task3]);
console.log(res);
// all 方法
Promise.all = function (taskArr) {
  return new Promise((resolve, reject) => {
    const resArr = [];
    let num = 0;

    // 遍历
    taskArr.forEach((item, index) => {
      item.then(
        (data) => {
          // 按下标存储数据
          resArr[index] = data;
          num++;

          if (num === taskArr.length) {
            resolve(resArr);
          }
        },
        (err) => {
          reject(err);
        }
      );
    });
  });
};

Class 的写法​

// 1. 定义Promise的类
class Promise {
  // 1.1  设置默认值
  PromiseState = "pendding";
  PromiseResult = undefined;
  callbacks = [];

  // 1.2 声明构造器
  constructor(executor) {
    // 1.3 声明成功或者失败的回调
    const success = (val) => {
      if (this.PromiseState !== "pendding") return;
      this.PromiseState = "fulfilled";
      this.PromiseResult = val;
    };
    const error = (err) => {
      if (this.PromiseState !== "pendding") return;
      this.PromiseState = "rejected";
      this.PromiseResult = err;
    };

    // 1.4 同步执行回调
    try {
      executor?.(success, error);
    } catch (err) {
      error(err.message || err);
    }
  }

  // 1.5 声明实例方法
  then(onFulfilled, onRejected) {
    return new Promise((resolve, reject) => {
      // 两个参数没有值得情况
      if (typeof onFulfilled !== "function") {
        onFulfilled = (v) => v;
      }
      if (typeof onRejected !== "function") {
        onRejected = (err) => {
          throw err;
        };
      }

      // 1.5.1 封装回调的处理
      const handler = (type) => {
        try {
          const result = type(this.PromiseResult);
          if (result instanceof Promise) {
            result.then(
              (data) => resolve(data),
              (err) => reject(err)
            );
          } else {
            resolve(result);
          }
        } catch (err) {
          reject(err);
        }
      };

      // 1.5.2 根据状态判断执行
      if (this.PromiseState === "pendding") {
        this.callbacks.push({
          ok() {
            handler(onFulfilled);
          },
          ng() {
            handler(onRejected);
          },
        });
      }
      if (this.PromiseState === "fulfilled") {
        setTimeout(() => handler(onFulfilled));
      }
      if (this.PromiseState === "rejected") {
        setTimeout(() => handler(onRejected));
      }
    });
  }

  // catch 方法需要考虑两个情景
  catch(onRejected) {
    return this.then(undefined, onRejected);
  }

  // 1.6 静态方法
  static resolve(data) {
    return new Promise((resolve, reject) => {
      if (data instanceof Promise) {
        data.then(resolve, reject);
      } else {
        resolve(data);
      }
    });
  }
  static reject(data) {
    return new Promise((resolve, reject) => {
      reject(data);
    });
  }

  // race  all
  static race(taskArr) {
    return new Promise((resolve, reject) => {
      taskArr.forEach((item) => {
        item.then(
          (data) => resolve(data),
          (err) => reject(err)
        );
      });
    });
  }

  static all(taskArr) {
    return new Promise((resolve, reject) => {
      // 声明最终的结果数组和成功的计时
      const resArr = [];
      let num = 0;

      taskArr.forEach((item, index) => {
        item.then(
          (data) => {
            // 每次成功会增加计时和按位置赋值
            num++;
            resArr[index] = data;

            // 证明成功的次数和任务的长度一致
            if (num === taskArr.length) {
              resolve(resArr);
            }
          },
          (err) => {
            reject(err);
          }
        );
      });
    });
  }
}
