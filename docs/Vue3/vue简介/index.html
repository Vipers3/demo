Vue 简介

1、直接在网页中使用 vue​

通过 script 标签引入 vue

<script src="https://unpkg.com/vue@3/dist/vue.global.js"></script>

创建一个根组件,在 vue3 中组件就是一个普通的 js 对象

<div id="root"></div>;

// 组件用来创建组件实例(vm),组件是组件实例的模板
// 组件 -->组件生成组件实例 --> 虚拟DOM --> DOM(在页面中呈现)
const Root = {
  data() {
    return {
      message: "Vue好棒!", //data方法返回的对象，其中的属性会自动的添加到组件实例中
    };
  }, //data是一个函数，需要一个对象作为返回值

  // 在模板中可以直接访问组件实例中的属性
  // 在模板中可以通过{{属性名}}来访问实例中的属性
  template: "<h1>我爱Vue,{{message}}</h1>", //希望组件呈现的样子
};

//   创建app实例
const app = Vue.createApp(Root);
// 将实例在页面中挂载
app.mount("#root");

// 创建实例并挂载
// 挂载到id为root的元素下
Vue.createApp(Root).mount("#root");

按钮案例​

创建一个按钮，可以显示按钮的点击次数

<div id="root">
  <!-- <button @click="count++">按钮</button>-点了{{count}}次 -->
</div>
<p>我是网页中的内容</p>
<!--如果直接将模板定义到网页中，此时模板必须复合html的规范  
    如果在组件中定义了template，则会优先使用template作为模板，同时根元素中的所有内容，都会被替换
    如果在组件中没有定义template，则会使用根元素的innerHTML作为模板
-->
const app ={
    data(){
        return{
            count:0;//记录点击的次数
            // data中的数据会自动和使用它的视图绑定，数据发送变化视图会自动刷新
        }
    },
    template:"<button>点我一下</button><span>{{count}}</span>"
}

2.使用 vite 创建​

vue 基本框架​

yarn init -y # 初始化项目
yarn add -D vite #安装vite
yarn add vue #安装vue
#注意：项目名要符合命名规范，否则会安装失败

创建 src 文件夹 用来存放 js 文件

在项目下创建一个 index.html 入口文件

在 index.html 中引入 index.js

<script type="module" src="./src/index.js"></script>

<div id="app"></div>

在 index.js 中引入 vue

// 这里引用的vue，默认不支持template属性来设置模板
// 需要引用vue/dist/vue.esm-bundler.js支持template模板
import { createApp } from "vue/dist/vue.esm-bundler.js";
// 创建组件
const App = {
  data() {
    return {
      msg: "今天天气真不错",
    };
  },
  template: `<h1>{{msg}}</h1>`,
};
// 创建应用挂载到页面

createApp(App).mount("#app");

启动 vue 项目

yarn vite --open

单文件组件​

template 是用字符串的形式在编写模板

  • 这些字符串会在项目运行时，在浏览器中会被编译为 js 函数
  • 在字符串中编写代码，体验很差

为了解决这个问题，Vue 为我们提供了一种单文件组件（SFC）

  • 单文件组件的格式是 vue
  • vue 文件用来编写单文件组件，vue 本身并不能被浏览器所识别所以它必须要被构建工具打包后，才可以使用
  • 同时 vue 文件在打包时，构建工具会直接将 template 转换为函数

浏览器无法识别 vue 格式的文件

yarn add @vitejs/plugin-vue -D

创建 vite.config.js 配置组件

import vue from "@vitejs/plugin-vue";
export default {
  plugins: [vue()],
};

子组件 Mybutton.vue

<script>
  export default {
    data() {
      return {
        count: 0,
      };
    },
  };
</script>
<template>
  <button @click="count++">点我一下</button><span>{{ count }}</span>
</template>

根组件 App.vue

<script>
  // 引入子组件
  import MyButton from "./components/MyButton.vue";
  /* 编写组件代码 */

  export default {
    data() {
      return {
        msg: "我爱vue",
      };
    },
    // 在组件中注册子组件
    components: {
      // MyButton:MyButton,//属性值和属性名相同可以省略
      // MyButton,  //模板在html中会被解析为小写 所以<MyButton></MyButton> 写在html不好使
      MyButton,
    },
  };
</script>

<template>
  <h1>{{msg}}</h1>
  <MyButton></MyButton>
</template>

主文件 index.js

// 引入vue
import { createApp } from "vue";
// 引入根组件
import App from "./App.vue";

// 创建应用并挂载
// mount()的返回值是根组件的实例
const vm = createApp(App).mount("#root");
